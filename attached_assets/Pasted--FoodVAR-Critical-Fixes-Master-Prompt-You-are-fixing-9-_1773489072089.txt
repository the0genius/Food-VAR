## FIX 1: Graceful Shutdown

**File:** `server/index.ts`

After the `server.listen(...)` block inside the async IIFE, add a graceful shutdown handler:

```ts
const shutdown = async (signal: string) => {
  logger.info(`${signal} received. Starting graceful shutdown...`);
  server.close(() => {
    logger.info("HTTP server closed");
    pool.end().then(() => {
      logger.info("DB pool drained");
      process.exit(0);
    }).catch(() => process.exit(1));
  });
  setTimeout(() => {
    logger.error("Forced shutdown after timeout");
    process.exit(1);
  }, 15000);
};

process.on("SIGTERM", () => shutdown("SIGTERM"));
process.on("SIGINT", () => shutdown("SIGINT"));
```

This requires importing `pool` from `./db`. Modify `server/db.ts` to export the pool:

```ts
export const pool = new pg.Pool({ ... });
export const db = drizzle(pool);
```

Then in `server/index.ts`, change the import to:
```ts
import { pool } from "./db";
```

The pool reference is needed by shutdown. The `db` import used elsewhere in routes still works because routes.ts imports from `./db` directly.

## FIX 2: DB Pool Configuration

**File:** `server/db.ts`

Replace the entire file with:

```ts
import { drizzle } from "drizzle-orm/node-postgres";
import pg from "pg";

if (!process.env.DATABASE_URL) {
  throw new Error("DATABASE_URL must be set");
}

const isProduction = process.env.NODE_ENV === "production";

export const pool = new pg.Pool({
  connectionString: process.env.DATABASE_URL,
  max: parseInt(process.env.DB_POOL_MAX || "20", 10),
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 5000,
  ...(isProduction && {
    ssl: {
      rejectUnauthorized: process.env.DB_SSL_REJECT_UNAUTHORIZED !== "false",
    },
  }),
});

pool.on("error", (err) => {
  console.error("Unexpected DB pool error:", err.message);
});

export const db = drizzle(pool);
```

Also add these to `.env.example` under the `# === Required ===` section:

```
# DB_POOL_MAX=20
# DB_SSL_REJECT_UNAUTHORIZED=true
```

**File:** `lib/query-client.ts` — make `getApiUrl()` work outside Replit:

Replace:
```ts
export function getApiUrl(): string {
  let host = process.env.EXPO_PUBLIC_DOMAIN;

  if (!host) {
    throw new Error("EXPO_PUBLIC_DOMAIN is not set");
  }

  let url = new URL(`https://${host}`);

  return url.href;
}
```

With:
```ts
export function getApiUrl(): string {
  const domain = process.env.EXPO_PUBLIC_API_URL || process.env.EXPO_PUBLIC_DOMAIN;

  if (!domain) {
    throw new Error("EXPO_PUBLIC_API_URL or EXPO_PUBLIC_DOMAIN must be set");
  }

  // Support both full URLs and bare domains
  if (domain.startsWith("http://") || domain.startsWith("https://")) {
    return domain.endsWith("/") ? domain : `${domain}/`;
  }

  return `https://${domain}/`;
}
```
